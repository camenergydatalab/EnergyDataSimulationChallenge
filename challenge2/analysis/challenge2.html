#visualization of the data-set as values per day

#calculate energy consumption 
date = pd.date_range('2011-04-18', '2011-05-24')
energy_part = np.zeros([6, len(date)])
for i, one_date in enumerate(date[:]):
    one_date = one_date.strftime('%Y-%m-%d')
    energy_part[0,i] = df_total_watt[one_date:one_date+' 03:00:00'].sum()
    energy_part[1,i] = df_total_watt[one_date+' 09:00:00':one_date+' 15:00:00'].sum()
    energy_part[2,i] = df_total_watt[one_date+' 21:00:00':one_date].sum()
    energy_part[3,i] = df_total_watt[one_date].sum()
    energy_part[4,i] = df_total_watt[one_date+' 03:00:00':one_date+' 09:00:00'].sum()
    energy_part[5,i] = df_total_watt[one_date+' 15:00:00':one_date+' 21:00:00'].sum()
    
#plot the average of energy consumption per day(average of daily, 9:00-15:00, and 21:00-3:00)
plt.figure(figsize=(16, 4))    
ind = np.arange(len(date))
plt.bar(ind, (energy_part[1]/6), 0.3, color="r", alpha=0.3, align="center", label="9:00-15:00")
plt.bar(ind+0.3, (energy_part[0]+energy_part[2])/6, 0.3, color="b", alpha=0.3, align="center", label="21:00:3:00")
plt.plot(ind, (energy_part[3])/24, color='k', lw=2, label="day")
plt.xlabel('Day')
plt.ylabel('The Average of Energy Consumption (W)')
plt.xticks(ind+0.15, [dt.strftime(i, '%m/%d') for i in date], rotation=50)
plt.legend( loc='upper left')
plt.autoscale(enable=True, axis='x',tight=True)

plt.show()

In this chart, the average of energy consumption per day is described as black line. To confirm the first feature, the average around noon(9:00-15:00) and midnight(21:00-3:00) are visualized with bar chart, respectively. From this chart, it is confirmed that although the most average around noon or midnight is larger than the daily average, it is the other way around in some days(4/23(Sat), 4/30(Sat), 5/7(Sat), 5/20(Fri)) .

#visualization of the data-set as clusters

#resample data-set from per 30mins to per day
df_daily = df_total_watt.resample('D', how='mean')

#clustering by kmeans
kmeans = KMeans(n_clusters=3)
kmeans.fit([[x] for x in df_daily.Energy.fillna(0).tolist()])
df_daily['label']=kmeans.labels_    

#sorting label
label_sorted = sorted([df_daily.ix[df_daily.label == i].ix[0][0] for i in range(3)])
color_sorted = [0]*3
colors = ['green', 'yellow', 'red']
for i, label in enumerate(map(lambda x: df_daily.ix[df_daily.Energy == x].ix[0][1], label_sorted)):
    color_sorted[int(label)]=colors[i]

#plot the resampled average of energy consumption with bar chart 
plt.figure(figsize=(16, 6))
plt.plot(df_daily.Energy.fillna(0), marker="o")
for row in range(df_daily.label.count()):
    color = color_sorted[int(df_daily.iloc[[row]].label)]
    plt.bar(float(row)-0.1, df_daily.iloc[[row]].Energy.fillna(0), 0.18, color=color, alpha=0.5)
plt.xticks(ind+0.15, [dt.strftime(i, '%m/%d') for i in date], rotation=50)
plt.xlabel('Day')
plt.ylabel('The Average of Energy Consumption (W)')
plt.autoscale(enable=True, axis='x',tight=True)
p = [0]*3
for i in range(3):
    p[i] = plt.Rectangle((0, 0), 1, 1, fc=colors[i], alpha=0.5)
plt.legend([p[2],p[1],p[0]], ["high","medium","low"],  loc='upper left')

#highlight the each weekends
plt.axvspan(5, 6, facecolor='k', alpha=0.1)
plt.axvspan(12, 13, facecolor='k', alpha=0.1)
plt.axvspan(19, 20, facecolor='k', alpha=0.1)
plt.axvspan(26, 27, facecolor='k', alpha=0.1)
plt.axvspan(33, 34, facecolor='k', alpha=0.1)

plt.show()

In this chart, the average of energy consumption per day is described with clustered information. Grey area highlights each weekends. From this chart, it is shown that the variation of energy consumption becomes large around each weekends.
